def update_host(opts)
  path = HOST_API_PATH
  if opts && opts[:id]
    id = opts.delete(:id)
    path = "#{HOST_API_PATH}/#{id}"
  end
  json_to_mdm_object(self.put_data(path, opts), HOST_MDM_CLASS, [])
end
